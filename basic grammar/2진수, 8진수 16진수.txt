# 2진수 : 0b
# 8진수 : 0o
# 16진수 : 0x

# bin(42) -> 42의 2진수 : 0b101010
# oct(42) -> 42의 8진수 : 0o52
# hex(42) -> 42의 16진수 : 0x2a

# int('0b101010', 2) -> 42
# int('0o52', 8) -> 42
# int('0x2a', 16) -> 42

# format(42, 'b') -> '101010'
# format(42, 'o') -> '52'
# format(42, 'x') -> '2a'
